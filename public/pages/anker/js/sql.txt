select t1.`id`, t1.`number`, t1.`cn_name`, t1.`name`, t2.`path`, t2.id as t2ID from user_user t1 left join
(
	select * from `user_user_accessory` a  where not exists(
		select 1 from user_user_accessory where fid=a.fid and id>a.id
	) order by fid
) as t2 on t1.id = t2.fid where t1.`locked` <> 1 order by t1.id
